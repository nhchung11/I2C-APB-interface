1  module i2c_master(
     2  	// i2c core signal
     3  	input wire clk,
     4  	input wire i2c_reset_n,
     5  	input wire [6:0] addr,
     6  	input wire [7:0] i2c_data_in,
     7  	input wire i2c_enable,
     8  	input wire rw,
     9  	output reg [7:0] i2c_data_out,
     10  	output wire i2c_ready,
     11  	inout i2c_sda,
     12  	inout wire i2c_scl,
     13  	input i2c_repeat_start,
     14  	input fifo_tx_empty,
     15  	input fifo_rx_full,
     16  	//control fifo
     17  	output reg fifo_tx_rd_en,
     18  	output reg fifo_rx_wr_en 
     19  	);
     20  
     21  	localparam IDLE = 0;
     22  	localparam START = 1;
     23  	localparam ADDRESS = 2;
     24  	localparam READ_ACK = 3;
     25  	localparam WRITE_DATA = 4;
     26  	localparam WRITE_ACK = 5;
     27  	localparam READ_DATA = 6;
     28  	localparam READ_ACK2 = 7;
     29  	localparam STOP = 8;
     30  	
     31  	localparam DIVIDE_BY = 4;
     32  
     33  	reg [7:0] state;
     34  	reg [7:0] saved_addr;
     35  	reg [7:0] counter;
     36  	reg [7:0] counter2 = 0;
     37  	reg write_enable;
     38  	reg sda_out;
     39  	reg i2c_scl_enable = 0;
     40  	reg i2c_clk = 1;
     41  	reg status;
     42  	reg count_done;
     43  
     44  	assign i2c_ready = ((i2c_reset_n == 1) && (state == IDLE)) ? 1 : 0;
     45  	assign i2c_scl = (i2c_scl_enable == 0 ) ? 1 : i2c_clk;
     46  	assign i2c_sda = (write_enable == 1) ? sda_out : 'bz;
     47  	pullup(i2c_sda);
     48  
     49  	always @(posedge clk) begin
     50  		if (counter2 == (DIVIDE_BY/2) - 1) begin
     51  			i2c_clk <= ~i2c_clk;
     52  			counter2 <= 0;
     53  		end
     54  		else counter2 <= counter2 + 1;
     55  	end 
     56  	
     57  	always @(negedge i2c_clk, negedge i2c_reset_n) begin
     58  		if(!i2c_reset_n) begin
     59  			i2c_scl_enable <= 0;
     60  		end else begin
     61  			if ((state == IDLE) || (state == START) || (state == STOP)) begin
     62  				i2c_scl_enable <= 0;
     63  			end else begin
     64  				i2c_scl_enable <= 1;
     65  			end
     66  		end
     67  	
     68  	end
     69  
     70  	always @(posedge clk) begin
     71  		case(state)
     72  			READ_ACK: begin
     73  					if (i2c_sda == 0) begin
     74  						fifo_tx_rd_en <= 1;
     75  						status <= 1;
     76  					end
     77  					if (status == 1) begin
     78  						fifo_tx_rd_en <= 0;
     79  					end
     80  				end
     81  			WRITE_DATA: status <= 0;
     82  			READ_ACK2: begin
     83  				if (i2c_sda == 0 && !fifo_tx_empty) begin
     84  						fifo_tx_rd_en <= 1;
     85  						status <= 1;
     86  					end
     87  					if (status == 1) begin
     88  						fifo_tx_rd_en <= 0;
     89  					end
     90  			end
     91  		endcase
     92  	end 
     93  	
     94  	always @(posedge i2c_clk, negedge i2c_reset_n) begin
     95  		if(!i2c_reset_n) begin
     96  			state <= IDLE;
     97  			fifo_tx_rd_en <= 0;
     98  			fifo_rx_wr_en <= 0;
     99  			count_done <= 0;
     100  		end		
     101  		else begin
     102  			case(state)
     103  				IDLE: begin
     104  					if (i2c_enable || i2c_repeat_start) begin
     105  						state <= START;
     106  						saved_addr <= {addr, rw};
     107  					end
     108  					else state <= IDLE;
     109  				end
     110  
     111  				START: begin
     112  					counter <= 7;
     113  					state <= ADDRESS;
     114  				end
     115  
     116  				ADDRESS: begin
     117  					if (counter == 0) begin 
     118  						state <= READ_ACK;
     119  					end else counter <= counter - 1;
     120  					
     121  				end
     122  
     123  				READ_ACK: begin
     124  					if (i2c_sda == 0) begin
     125  						counter <= 7;
     126  						if(saved_addr[0] == 0) state <= WRITE_DATA;
     127  						else state <= READ_DATA;
     128  					end else state <= STOP;
     129  				end
     130  
     131  				WRITE_DATA: begin
     132  					if(counter == 0) begin
     133  						state <= READ_ACK2;
     134  					end else counter <= counter - 1;
     135  				end
     136  				
     137  				READ_ACK2: begin
     138  					if (i2c_sda == 0 && i2c_repeat_start) begin
     139  						state <= IDLE;
     140  					end
     141  					else if (i2c_sda == 0) begin
     142  							state <= WRITE_DATA;
     143  							counter <= 7;	
     144  							if(fifo_tx_empty) count_done <= 1;
     145  							if(count_done) begin
     146  								count_done <= 0;
     147  								state <= STOP;
     148  							end
     149  						end
     150  					else begin
     151  						state <= STOP;
     152  					end
     153  				end
     154  
     155  				READ_DATA: begin
     156  					i2c_data_out[counter] <= i2c_sda;
     157  					if (counter == 0) begin
     158  						state <= WRITE_ACK;
     159  						fifo_rx_wr_en <= 1;
     160  					end
     161  					else counter <= counter - 1;
     162  				end
     163  				
     164  				WRITE_ACK: begin
     165  					fifo_rx_wr_en <= 0;
     166  					if (i2c_sda == 0 && i2c_repeat_start) begin
     167  						state <= IDLE;
     168  					end
     169  					else if (i2c_sda == 0) begin
     170  							state <= READ_DATA;
     171  							counter <= 7;	
     172  							if(fifo_rx_full) count_done <= 1;
     173  							if(count_done) begin
     174  								count_done <= 0;
     175  								state <= STOP;
     176  							end
     177  						end
     178  					else begin
     179  						state <= STOP;
     180  					end
     181  				end
     182  
     183  				STOP: begin
     184  					state <= IDLE;
     185  				end
     186  			endcase
     187  		end
     188  	end
     189  	
     190  	always @(negedge i2c_clk, negedge i2c_reset_n) begin
     191  		if(!i2c_reset_n) begin
     192  			write_enable <= 1;
     193  			sda_out <= 1;
     194  		end else begin
     195  			case(state)
     196  				
     197  				START: begin
     198  					write_enable <= 1;
     199  					sda_out <= 0;
     200  				end
     201  				
     202  				ADDRESS: begin
     203  					sda_out <= saved_addr[counter];
     204  				end
     205  				
     206  				READ_ACK: begin
     207  					write_enable <= 0;
     208  				end
     209  				
     210  				WRITE_DATA: begin 
     211  					write_enable <= 1;
     212  					sda_out <= i2c_data_in[counter];
     213  				end
     214  				READ_ACK2: begin
     215  					write_enable <= 0;				
     216  				end
     217  				
     218  				WRITE_ACK: begin
     219  					write_enable <= 1;
     220  					sda_out <= 0;
     221  				end
     222  				
     223  				READ_DATA: begin
     224  					write_enable <= 0;				
     225  				end
     226  				
     227  				STOP: begin
     228  					write_enable <= 1;
     229  					sda_out <= 1;
     230  				end
     231  			endcase
     232  		end
     233  	end
     234  	
     235  endmodule
